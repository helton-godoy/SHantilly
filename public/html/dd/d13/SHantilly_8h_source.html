174struct DialogCommand {
175 DialogCommand() {
176 buffer[BUFFER_SIZE - 1] = '\0';
177 title = name = text = auxtext = BUFFER_SIZE - 1;
178 }
179
180 char *getTitle() { return buffer + title; }
181 char *getName() { return buffer + name; }
182 char *getText() { return buffer + text; }
183 char *getAuxText() { return buffer + auxtext; }
184
185 char buffer[BUFFER_SIZE];
186 size_t title;
187 size_t name;
188 size_t text;
189 size_t auxtext;
190
191 unsigned int command;
192 unsigned int control;
193};
