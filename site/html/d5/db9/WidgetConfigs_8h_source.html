32struct ActionConfig {
33 enum Type { Shell, Script, Set, Query, Callback };
34
35 Type type = Shell;
36 QString command; // Para Shell/Script: o comando ou caminho
37 QString targetWidget; // Para Set/Query: nome do widget alvo
38 QString property; // Para Set: propriedade a modificar
39 QString value; // Para Set: novo valor
40 QString variable; // Para Query: nome da variável de destino
41
42 bool isValid() const {
43 switch (type) {
44 case Shell:
45 case Script:
46 case Callback:
47 return !command.isEmpty();
48 case Set:
49 return !targetWidget.isEmpty() && !property.isEmpty();
50 case Query:
51 return !targetWidget.isEmpty() && !variable.isEmpty();
52 }
53 return false;
54 }
55};
